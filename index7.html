Linux Users

Tue 13 August 2013

[cc lang='bash']
# show defaults
\$ useradd -D
# add temp user with pass temp
\$ useradd --comment 'Temp User' --create-home --gid 100 --uid 10000 --password eCGwS26XXJXak --shell /bin/bash temp

# lock the password
\$ passwd --lock temp
# unlock the password
\$ passwd --unlock temp

# remove user and home
\$ userdel --remove temp
[/cc]
